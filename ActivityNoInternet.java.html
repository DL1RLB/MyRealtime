36                       .setCancelable(false)
37                       .setMessage(getString(R.string.NoInternet))
38                       .setPositiveButton("Verstanden", (dialog, which) -> {
39                           Intent i2 = new Intent(ActivityNoInternet.this, MainActivity.class); // diese Seite öffnen wenn kein Internet gefunden wird
40                           startActivityForResult(i2, intencode); // starten
41                       })
42                       .setNegativeButton("Ja", (dialog, which) -> {
43                           Intent i3 = new Intent(ActivityNoInternet.this, ActivityInfo.class); // diese Seite öffnen wenn auf Ja geklickt wurde
44                           startActivityForResult(i3, intencode); // start
